$region = "East US"  # Replace with your desired region
$desiredSkus = @("Standard_DS2_v2", "Standard_D2_v3", "Standard_F2s_v2")  # Replace with your desired list of VM SKUs

$availableSkus = Get-AzComputeResourceSku | Where-Object { $_.Locations -contains $region }

$unavailableSkus = @()

foreach ($sku in $desiredSkus) {
    if ($availableSkus.Name -notcontains $sku) {
        $unavailableSkus += $sku
    }
}

if ($unavailableSkus.Count -gt 0) {
    # Trigger an alert or perform desired action
}
