# Get the quota information for a specific scope
# Replace "subscriptions/xxx/providers/Microsoft.Compute/locations/eastus" with the appropriate scope
$x = Get-AzQuota -Scope "subscriptions/xxx/providers/Microsoft.Compute/locations/eastus" |
    Where-Object { $_.NameLocalizedValue -eq "Standard EIDv5 Family vCPUs" }

# Output the name of the quota
$x.name
