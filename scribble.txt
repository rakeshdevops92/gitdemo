# Define the inline script content
$scriptContent = @'
#!/bin/bash
echo "Hello from inline script on the Azure Automation runbook!"
'@

# Save the inline script to a temporary path
$scriptLocalPath = "D:\Temp\yourScript.sh"
Set-Content -Path $scriptLocalPath -Value $scriptContent

# Run your script on the VM
Invoke-AzVMRunCommand -ResourceGroupName $ResourceGroupName -Name $VMName -CommandId 'RunShellScript' -ScriptPath $scriptLocalPath
